    # File app/models/ptainstructor.rb, line 109
def formatted_number(number)
  digits = number.gsub(%r\D/, '').split(%r/)

  if (digits.length == 10)
    digits = digits.join
    '(%s) %s-%s' % [ digits[0,3], digits[3,3], digits[6,4] ]
  end
end

    # File app/models/ptainstructor.rb, line 77
def phone_is_valid?
  return false unless not_nil_and_string(self.phone)
  return false unless ((self.phone =~ %r^\(?([0-9]{3})\)?[-. ]?([0-9]{3})[-. ]?([0-9]{4})$/) == 0)
  self.phone = formatted_number(self.phone)
  return true
end
