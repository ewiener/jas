def create
  #check for admin
  @semester = Semester.find_by_id params[:semester_id]
  return unless semester_is_valid(@semester)

  @instructor = @semester.instructors.create(params[:instructor])
  if @instructor.new_record?
    flash[:warning] = @instructor.errors
    flash[:instructor] = @instructor
    redirect_to new_semester_instructor_path(@semester)
    return
  else
    flash[:notice] = "#{@instructor.first_name} #{@instructor.last_name} was successfully added to the database."
    redirect_to semester_instructors_path(@semester)
  end
end

def destroy
  #Check is the instructor is an admin.  Only admins should be able to delete instructors
  @instructor = Instructor.find_by_id params[:id]
  return unless instructor_is_valid(@instructor)

  first_name = @instructor.first_name
  last_name = @instructor.last_name

  if @instructor.destroy
    flash[:notice] = "#{first_name} #{last_name} was successfully deleted."
  else
    flash[:warning] = @instructor.errors
  end

  redirect_to semester_instructors_path
end

def index
  if params[:semester_id]
    @semester = Semester.find_by_id params[:semester_id]
    if not @semester
      flash[:warning] = [[:semester_id, "Unable to locate the specified semester."]]
      redirect_to semesters_path
      return
    end
  else
    flash[:warning] = [[:semester_id, "No semester was given."]]
    redirect_to semesters_path
    return
  end
  @instructors = Instructor.find_all_by_semester_id(@semester, :order => "first_name asc, last_name asc")
end

def instructor_is_valid(user)
  if(user == nil)
    flash[:warning] = [[:id, "Could not find the corresponding PTA instructor."]]
    redirect_to semester_instructors_path
    return false
  end
  return true
end

def update
  #Check if admin or if the current instructor matches the id of the instructor being modified
  @semester = Semester.find_by_id params[:semester_id]
  @instructor = Instructor.find_by_id params[:id]
  return unless instructor_is_valid(@instructor)

  if @instructor.update_attributes(params[:instructor])
    flash[:notice] = "#{@instructor.first_name} #{@instructor.last_name}'s information was successfully updated."
    redirect_to semester_instructors_path(@semester)
  else
    flash[:warning] = @instructor.errors
    redirect_to edit_semester_instructor_path(@semester,@instructor)
  end
end
