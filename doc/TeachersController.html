def create
  @semester = Semester.find_by_id params[:semester_id]
  return unless semester_is_valid(@semester,"Error: Unable to find a semester to associated with the classroom.")

  @classroom = @semester.classrooms.create(params[:classroom])
  if @classroom.new_record?
    flash[:warning] = @classroom.errors
    flash[:classroom] = @classroom
    redirect_to new_semester_classroom_path(@semester)
    return
  else
    flash[:notice] = "Successfully added #{@classroom.teacher} to the database."
    redirect_to semester_classrooms_path(@semester)
  end
end

def destroy
  @classroom = Classroom.find_by_id(params[:id])
  return unless classroom_is_valid(@classroom)

  name = @classroom.name

  if @classroom.destroy
    flash[:notice] = "#{name} was successfully deleted from the database."
  else
    flash[:warning] = @classroom.errors
  end

  redirect_to semester_classrooms_path
end

def index
  @semester = Semester.find_by_id params[:semester_id]
  return unless semester_is_valid(@semester)
  classrooms1 = Classroom.where("semester_id = ? AND grade = ?", @semester.id, "K").order("name asc")
  classrooms2 = Classroom.where("semester_id = ? AND grade != ?", @semester.id, "K").order("grade asc","name asc")
  @classrooms = classrooms1 + classrooms2
end

def update
  @semester = Semester.find_by_id params[:semester_id]
  return unless semester_is_valid(@semester)
  @classroom = Classroom.find_by_id params[:id]
  return unless classroom_is_valid(@classroom)

  if @classroom.update_attributes(params[:classroom])
    flash[:notice] = "#{@classroom.teacher}'s entry was successfully updated."
    redirect_to semester_classrooms_path(@semester)
  else
    flash[:warning] = @classroom.errors
    redirect_to edit_semester_classroom_path(@semester, @classroom)
  end
end
